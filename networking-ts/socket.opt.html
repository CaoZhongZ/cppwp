18.5.1 Class socket_base::linger [socket.opt.linger]

1

The linger class represents a socket option for controlling the behavior when a socket is closed and unsent data is present.

namespace std {
namespace experimental {
namespace net {
inline namespace v1 {

  class socket_base::linger
  {
  public:
    // constructors:
    linger() noexcept;
    linger(bool e, chrono::seconds t) noexcept;

    // members:
    bool enabled() const noexcept;
    void enabled(bool e) noexcept;

    chrono::seconds timeout() const noexcept;
    void timeout(chrono::seconds t) noexcept;
  };

} // inline namespace v1
} // namespace net
} // namespace experimental
} // namespace std
2

linger satisfies the requirements of Destructible (C++Std §[destructible]), DefaultConstructible (C++Std §[defaultconstructible]), CopyConstructible (C++Std §[copyconstructible]), CopyAssignable (C++Std §[copyassignable]), GettableSocketOption ([socket.reqmts.gettablesocketoption]), and SettableSocketOption ([socket.reqmts.settablesocketoption]).
